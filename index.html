Simple Integration

Add hot reload to any iOS app with just a few lines

SwiftUI App Swift 
import SwiftUI
import SwiftMetro

@main
struct MyApp: App {
    init() {
        // Enable hot reload in debug builds
        #if DEBUG
        SwiftMetro.shared.start()
        #endif
    }
    
    var body: some Scene {
        WindowGroup {
            ContentView()
                .enableHotReload()
        }
    }
}
